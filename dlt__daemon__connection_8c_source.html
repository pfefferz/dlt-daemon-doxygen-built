67 STATIC int dlt_connection_send(DltConnection *conn,
68  void *msg,
69  size_t msg_size)
70 {
71  DltConnectionType type = DLT_CONNECTION_TYPE_MAX;
72 
73  if ((conn != NULL) && (conn->receiver != NULL))
74  {
75  type = conn->type;
76  }
77 
78  switch (type)
79  {
80  case DLT_CONNECTION_CLIENT_MSG_SERIAL:
81  if(write(conn->receiver->fd, msg, msg_size) > 0)
82  return DLT_DAEMON_ERROR_OK;
83  return DLT_DAEMON_ERROR_UNKNOWN;
84  case DLT_CONNECTION_CLIENT_MSG_TCP:
85  return dlt_daemon_socket_sendreliable(conn->receiver->fd, msg, msg_size);
86  default:
87  return DLT_DAEMON_ERROR_UNKNOWN;
88  }
89 }

193 STATIC DltReceiver *dlt_connection_get_receiver(DltDaemonLocal *daemon_local,
194  DltConnectionType type,
195  int fd)
196 {
197  DltReceiver *ret = NULL;
198 
199  switch (type)
200  {
201  case DLT_CONNECTION_CONTROL_CONNECT:
202  /* FALL THROUGH */
203  case DLT_CONNECTION_CONTROL_MSG:
204  /* FALL THROUGH */
205  case DLT_CONNECTION_CLIENT_CONNECT:
206  /* FALL THROUGH */
207  case DLT_CONNECTION_CLIENT_MSG_TCP:
208  ret = calloc(1, sizeof(DltReceiver));
209  if (ret) {
210  dlt_receiver_init(ret, fd, DLT_DAEMON_RCVBUFSIZESOCK);
211  }
212  break;
213  case DLT_CONNECTION_CLIENT_MSG_SERIAL:
214  ret = calloc(1, sizeof(DltReceiver));
215  if (ret) {
216  dlt_receiver_init(ret, fd, DLT_DAEMON_RCVBUFSIZESERIAL);
217  }
218  break;
219  case DLT_CONNECTION_APP_MSG:
220  /* FALL THROUGH */
221  case DLT_CONNECTION_ONE_S_TIMER:
222  /* FALL THROUGH */
223  case DLT_CONNECTION_SIXTY_S_TIMER:
224 #ifdef DLT_SYSTEMD_WATCHDOG_ENABLE
225  /* FALL THROUGH */
226  case DLT_CONNECTION_SYSTEMD_TIMER:
227 #endif
228  /* FALL THROUGH */
229  case DLT_CONNECTION_GATEWAY_TIMER:
230  ret = calloc(1, sizeof(DltReceiver));
231  if (ret) {
232  dlt_receiver_init(ret, fd, DLT_DAEMON_RCVBUFSIZE);
233  }
234  break;
235  case DLT_CONNECTION_GATEWAY:
236  /* We rely on the gateway for init */
237  ret = dlt_gateway_get_connection_receiver(&daemon_local->pGateway, fd);
238  break;
239  default:
240  ret = NULL;
241  }
242 
243  return ret;
244 }

257 void *dlt_connection_get_callback(DltConnection *con)
258 {
259  void *ret = NULL;
260  DltConnectionType type = DLT_CONNECTION_TYPE_MAX;
261 
262  if (con)
263  {
264  type = con->type;
265  }
266 
267  switch (type)

270  ret = dlt_daemon_process_client_connect;
271  break;
272  case DLT_CONNECTION_CLIENT_MSG_TCP:
273  ret = dlt_daemon_process_client_messages;
274  break;
275  case DLT_CONNECTION_CLIENT_MSG_SERIAL:
276  ret = dlt_daemon_process_client_messages_serial;
277  break;
278  case DLT_CONNECTION_APP_MSG:
279  ret = dlt_daemon_process_user_messages;
280  break;
281  case DLT_CONNECTION_ONE_S_TIMER:
282  ret = dlt_daemon_process_one_s_timer;
283  break;
284  case DLT_CONNECTION_SIXTY_S_TIMER:
285  ret = dlt_daemon_process_sixty_s_timer;
286  break;
287 #ifdef DLT_SYSTEMD_WATCHDOG_ENABLE
288  case DLT_CONNECTION_SYSTEMD_TIMER:
289  ret = dlt_daemon_process_systemd_timer;
290  break;
291 #endif
292  case DLT_CONNECTION_CONTROL_CONNECT:
293  ret = dlt_daemon_process_control_connect;
294  break;
295  case DLT_CONNECTION_CONTROL_MSG:
296  ret = dlt_daemon_process_control_messages;

320 void dlt_connection_destroy(DltConnection *to_destroy)
321 {
322  to_destroy->id = 0;
323  close(to_destroy->receiver->fd);
324  dlt_connection_destroy_receiver(to_destroy);
325  /* connection pointer might be in epoll queue and used even after destroying
326  * it. To make sure it is not used anymore, connection type is invalidated */
327  to_destroy->type = DLT_CONNECTION_TYPE_MAX;
328  free(to_destroy);
329 }

346 int dlt_connection_create(DltDaemonLocal *daemon_local,
347  DltEventHandler *evh,
348  int fd,
349  int mask,
350  DltConnectionType type)
351 {
352  DltConnection *temp = NULL;
353 
354  if (fd < 0) {
355  /* Nothing to do */
356  return 0;
357  }
358 
359  if (dlt_event_handler_find_connection(evh, fd) != NULL)
360  {
361  /* No need for the same client to be registered twice
362  * for the same event.
363  * TODO: If another mask can be expected,
364  * we need it to update the epoll event here.
365  */
366  return 0;
367  }
368 
369  temp = (DltConnection *)malloc(sizeof(DltConnection));
370 
371  if (temp == NULL)
372  {
373  dlt_log(LOG_CRIT, "Allocation of client handle failed\n");
374  return -1;
375  }
376 
377  memset(temp, 0, sizeof(DltConnection));
378 
379  temp->receiver = dlt_connection_get_receiver(daemon_local, type, fd);
380  if (!temp->receiver) {
381  char local_str[DLT_DAEMON_TEXTBUFSIZE];
382  snprintf(local_str,
383  DLT_DAEMON_TEXTBUFSIZE,
384  "Unable to get receiver from %d connection.\n",
385  type);
386 
387  dlt_log(LOG_CRIT, local_str);
388  free(temp);
389  return -1;
390  }
391 
392  /* We are single threaded no need for protection. */
393  temp->id = connectionId++;
394  if (!temp->id)
395  {
396  /* Skipping 0 */
397  temp->id = connectionId++;
398  }
399 
400  temp->type = type;
401  temp->status = ACTIVE;
402 
403  /* Now give the ownership of the newly created connection
404  * to the event handler, by registering for events.
405  */
406  return dlt_event_handler_register_connection(evh, daemon_local, temp, mask);
407 }
