788 static inline DltReturnValue dlt_user_is_logLevel_enabled(DltContext *handle,DltLogLevelType loglevel)
789 {
790  if (handle == NULL || handle->log_level_ptr == NULL)
791  {
792  return DLT_RETURN_WRONG_PARAMETER;
793  }
794 
795  if (loglevel <= (DltLogLevelType)(*(handle->log_level_ptr)) && loglevel != DLT_LOG_OFF)
796  {
797  return DLT_RETURN_TRUE;
798  }
799 
800  return DLT_RETURN_LOGGING_DISABLED;
801 }
