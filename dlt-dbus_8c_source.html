54  if (!dbus_message_marshal(message,
55  buf,
56  &len_p))

58  fprintf (stderr, "Failed to serialize DBus message!\n");
59  return DBUS_HANDLER_RESULT_HANDLED;

63

72  DLT_UNREGISTER_APP ();
